// ==================== 初始化引导系统 ==================== const GUIDE_KEY = 'shenke_onboarded'; let onboardingActive = false; let onboardingStep = 0; let tempPositions = []; // 检查是否需要引导 function checkOnboarding() { const onboarded = localStorage.getItem(GUIDE_KEY); if (!onboarded) { // 延迟一点，让欢迎卡片先显示 setTimeout(startOnboarding, 1500); } } // 开始引导 function startOnboarding() { if (onboardingActive) return; onboardingActive = true; onboardingStep = 1; tempPositions = []; // 发送引导欢迎消息 const guideMsg = `📋 持仓初始化引导\n\n在开始使用深克之前，我想先了解一下你目前的持仓情况。这样我才能帮你做更精准的复盘和预警。\n\n你目前持有哪些股票？请告诉我股票名称，可以一次说多只，比如“茅台、宁德时代、比亚迪”。`; addMessage('ai', guideMsg); } // 处理引导流程中的用户输入 function handleOnboardingInput(text) { if (onboardingStep === 1) { // 解析股票名称 const stockNames = text.split(/[,，、\s]+/).filter(s => s.trim()); if (stockNames.length === 0) { addMessage('ai', '请至少输入一只股票的名称，比如“茅台”或“宁德时代”。'); return; } tempPositions = stockNames.map(name => ({ name: name.trim(), code: '', cost: null, quantity: null, stopProfit: null, stopLoss: null })); // 开始逐一询问成本价 onboardingStep = 2; currentPositionIndex = 0; askCostPrice(); return; } if (onboardingStep === 2) { // 解析成本价 const price = parseFloat(text.replace(/[^0-9.]/g, '')); if (isNaN(price) || price <= 0) { addMessage('ai', '请输入一个有效的数字，比如“32.50”。'); return; } tempPositions[currentPositionIndex].cost = price; onboardingStep = 3; askQuantity(); return; } if (onboardingStep === 3) { // 解析持仓数量 const quantity = parseInt(text.replace(/[^0-9]/g, '')); if (isNaN(quantity) || quantity <= 0) { addMessage('ai', '请输入一个有效的整数，比如“100”。'); return; } tempPositions[currentPositionIndex].quantity = quantity; // 默认止盈止损 tempPositions[currentPositionIndex].stopProfit = Math.round(tempPositions[currentPositionIndex].cost * 1.15 * 100) / 100; tempPositions[currentPositionIndex].stopLoss = Math.round(tempPositions[currentPositionIndex].cost * 0.9 * 100) / 100; // 检查是否还有下一只 currentPositionIndex++; if (currentPositionIndex < tempPositions.length) { onboardingStep = 2; askCostPrice(); } else { // 所有股票都填完了，显示确认信息 onboardingStep = 4; showConfirmSummary(); } return; } if (onboardingStep === 4) { // 确认或修改 if (text.includes('确认') || text.includes('没问题') || text.includes('可以') || text === 'ok' || text === 'OK') { completeOnboarding(); } else if (text.includes('修改') || text.includes('改') || text.includes('调整')) { addMessage('ai', '你想修改哪只股票的止盈止损线？请告诉我股票名称、止盈百分比和止损百分比，比如“茅台止盈20%止损8%”。'); onboardingStep = 5; } else { addMessage('ai', '请回复“确认”保存当前设置，或回复“修改”调整止盈止损线。'); } return; } if (onboardingStep === 5) { // 修改止盈止损 addMessage('ai', '收到，我会根据你的调整更新。目前修改功能需要重新输入，请先回复“确认”完成初始化，之后可以在对话中随时调整。'); onboardingStep = 4; } } let currentPositionIndex = 0; function askCostPrice() { const pos = tempPositions[currentPositionIndex]; addMessage('ai', `你持有 ${pos.name}，成本价是多少？比如“32.50”。`); } function askQuantity() { const pos = tempPositions[currentPositionIndex]; addMessage('ai', `你持有 ${pos.name} 多少股？比如“100”。`); } function showConfirmSummary() { let summary = `📊 持仓确认\n\n以下是根据你输入的信息生成的持仓列表，止盈线默认+15%，止损线默认-10%：\n\n`; tempPositions.forEach(pos => { summary += `📌 ${pos.name}\n 成本价：${pos.cost} | 数量：${pos.quantity}股\n 止盈：${pos.stopProfit} (+15%) | 止损：${pos.stopLoss} (-10%)\n\n`; }); summary += `回复“确认”保存设置，或回复“修改”调整止盈止损线。`; addMessage('ai', summary); } function completeOnboarding() { // 存入 localStorage localStorage.setItem(GUIDE_KEY, 'true'); localStorage.setItem('shenke_positions', JSON.stringify(tempPositions)); // 存入记忆桶（如果有的话） savePositionsToMemory(tempPositions); onboardingActive = false; onboardingStep = 0; tempPositions = []; addMessage('ai', '✅ 持仓初始化完成！\n\n我已经记住了你的持仓信息。从明天开始，深克会在每日复盘和预警中自动关联这些股票。\n\n你可以随时输入“查看我的持仓”来确认，或者输入“修改茅台止盈线到20%”来调整。\n\n祝投资顺利！'); } // 尝试存入记忆桶（如果 miyu-memory 已部署） async function savePositionsToMemory(positions) { try { await fetch('https://miyu-memory-shenke-wuwiiajgfl.cn-hangzhou.fcapp.run', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ project: { name: 'shenke', status: '用户持仓已初始化，共' + positions.length + '只股票' } }) });
